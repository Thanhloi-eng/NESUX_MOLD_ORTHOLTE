🗄 Tạo bảng Supabase (lần đầu)
Nếu gặp lỗi "Could not find the table", chạy SQL này trong Supabase → SQL Editor: 
-- Chạy trong Supabase > SQL Editor
CREATE TABLE IF NOT EXISTS public.scan_records (
  id bigserial PRIMARY KEY,
  mold_code text, rpro_code text, size_label text,
  qr_num text, qty_done int, qty_order int, pct int,
  worker_name text, shift_name text, scan_type text, rpro_group text,
  scanned_at timestamptz DEFAULT now()
);
-- Nếu bảng đã tồn tại, thêm cột shift_name (chạy an toàn):
ALTER TABLE public.scan_records ADD COLUMN IF NOT EXISTS shift_name text;
CREATE TABLE IF NOT EXISTS public.mold_requests (
  id bigserial PRIMARY KEY,
  mold_code text, size_label text, note text,
  worker_name text, status text DEFAULT 'pending',
  created_at timestamptz DEFAULT now(),
  resolved_at timestamptz
);
ALTER TABLE public.scan_records ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.mold_requests ENABLE ROW LEVEL SECURITY;
CREATE POLICY "anon_all_scan" ON public.scan_records FOR ALL TO anon USING (true) WITH CHECK (true);
CREATE POLICY "anon_all_req" ON public.mold_requests FOR ALL TO anon USING (true) WITH CHECK (true);

-- ═══ URGENT ORDERS (Đơn Gấp & Đơn Bù Gấp) ═══
CREATE TABLE IF NOT EXISTS public.urgent_orders (
  id bigserial PRIMARY KEY,
  rpro_codes text[],
  mold_codes text,
  order_type text,
  priority_pct int,
  note text,
  worker_id text, worker_name text, worker_title text,
  status text DEFAULT 'Chờ xử lý',
  accepted_id text, accepted_name text, accepted_title text,
  accepted_at timestamptz,
  rpro_done text[] DEFAULT '{}',
  created_at timestamptz DEFAULT now()
);
ALTER TABLE public.urgent_orders ENABLE ROW LEVEL SECURITY;
CREATE POLICY "anon_all_urgent" ON public.urgent_orders FOR ALL TO anon USING (true) WITH CHECK (true);
